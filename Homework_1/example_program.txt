.data
	TERMS:3
	result:0
	temp:0
	odd:0
.code
	ldi 1 TERMS
	ldi 0 result
	ldi 2 temp
	ldi 3 odd
	ld 0 0
	jz out
	jmp sum
square	add result result terms
	dec temp
	jz sum
	jmp square
sum	and odd terms 1
	jz oddgo
	jmp square
	dec terms
oddgo	jmp square
	dec terms
	dec terms
	jz terms
	jmp oddgo
out	ldi 4 result
	st 0 4
end	jmp end
